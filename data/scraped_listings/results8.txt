				function resetExpandables(selectorVal) {
					/*  */

					$(selectorVal).each(function(index) {
					var slideHeight = 250;
					    if ($(selectorVal).attr('class').split(' ')[index]=='heroRegionContent'){
					    		slideHeight=103;
					    };
					    var $this = $(this);
					    var $wrap = $this.children(".wrap");
					    var defHeight = $wrap.height();
					    if (defHeight >= slideHeight) {
					        var $readMore = $this.find(".read-more");
					        $wrap.css("height", slideHeight + "px");
					        $readMore.append("<a href='#'>More +</a>");
					        $readMore.children("a").bind("click", function(event) {
					            var curHeight = $wrap.height();
					            if (curHeight == slideHeight) {
					                $wrap.animate({
					                    height: defHeight
					                }, "normal");
					                $(this).text("Less -");
					                $readMore.children(".elipses").hide();
					                // $wrap.children(".gradient").fadeOut();
					            } else {
					                $wrap.animate({
					                    height: slideHeight
					                }, "normal");
					                $(this).text("More +");
					                $readMore.children(".elipses").show();
					                // $wrap.children(".gradient").fadeIn();
					            }
					            return false;
					        });

					    	$this.find(".read-more").hide();
